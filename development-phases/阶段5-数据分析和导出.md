  class AnalyticsService {
    constructor(logService) {

    async getUserAnalytics(userId, options = {}) {
      const {
        startDate = new Date(Date.now() - 30 * 24 * 60 * 60 * 1000), // 默认30天
        endDate = new Date(),
        includeDetails = false,
        realtime = false
      } = options;
      
      // 如果需要实时数据，跳过缓存
      if (!realtime) {
        const cacheKey = `${userId}_${startDate}_${endDate}`;
        if (this.aggregationCache.has(cacheKey)) {
          return this.aggregationCache.get(cacheKey);
        }
      }
      
      // 收集原始数据
      const rawData = await this.collectRawData(userId, startDate, endDate);
      
      // 执行多维度分析
      const analytics = {
        overview: await this.analyzeOverview(rawData),
        ctdp: await this.analyzeCTDP(rawData),
        rsip: await this.analyzeRSIP(rawData),
        behavioral: await this.analyzeBehavioral(rawData),

    async collectRawData(userId, startDate, endDate) {

    async analyzeOverview(rawData) {
      const { mainChains, auxChains, patternTrees, activityLogs } = rawData;
      
      // 计算总体统计
      const totalTasks = mainChains.reduce((sum, chain) => 
        sum + chain.nodes.length, 0
      );
      
      const completedTasks = mainChains.reduce((sum, chain) => 
        sum + chain.nodes.filter(n => n.status === 'completed').length, 0
      );
      
      const totalReservations = auxChains.reduce((sum, chain) => 
        sum + chain.nodes.length, 0
      );
      
      const executedReservations = auxChains.reduce((sum, chain) => 
        sum + chain.nodes.filter(n => n.status === 'executed').length, 0
      );
      
      const totalPatterns = patternTrees.reduce((sum, tree) => 
        sum + tree.nodes.length, 0
      );
      
      const activePatterns = patternTrees.reduce((sum, tree) => 
        sum + tree.nodes.filter(n => n.status === 'active').length, 0
      );
      
      // 计算总时间投入（分钟）
      const totalTimeInvested = mainChains.reduce((sum, chain) => {
        return sum + chain.nodes.reduce((nodeSum, node) => {
          if (node.endTime && node.startTime) {
            return nodeSum + (new Date(node.endTime) - new Date(node.startTime)) / 60000;
          }
          return nodeSum;
        }, 0);
      }, 0);
      
      // 计算成功率
      const taskSuccessRate = totalTasks > 0 ? completedTasks / totalTasks : 0;
      const reservationExecutionRate = totalReservations > 0 
        ? executedReservations / totalReservations : 0;
      
      // 计算活跃天数
      const activeDays = new Set(
        activityLogs.map(log => 
          new Date(log.timestamp).toDateString()
        )
      ).size;
      
      return {
        totalTasks,
        completedTasks,
        taskSuccessRate,
        totalReservations,
        executedReservations,
        reservationExecutionRate,
        totalPatterns,
        activePatterns,
        totalTimeInvested: Math.round(totalTimeInvested),
        averageTaskDuration: totalTasks > 0 
          ? Math.round(totalTimeInvested / totalTasks) : 0,
        activeDays,
        dailyAverage: {
          tasks: activeDays > 0 ? (totalTasks / activeDays).toFixed(1) : 0,
          timeInvested: activeDays > 0 
            ? Math.round(totalTimeInvested / activeDays) : 0
        }
      };
    }
    
    async analyzeCTDP(rawData) {
      const { mainChains, auxChains } = rawData;

          sum + chain.nodes.length, 0

          reservationEffectiveness.totalReservations;

      // 违规模式分析

        acc.push(...chain.violationPatterns);

    calculateMaxConsecutive(nodes) {
      let maxConsecutive = 0;
      let currentConsecutive = 0;
      
      nodes.forEach(node => {
        if (node.status === 'completed') {
          currentConsecutive++;
          maxConsecutive = Math.max(maxConsecutive, currentConsecutive);
        } else {
          currentConsecutive = 0;
        }
      });
      
      return maxConsecutive;
    }
    
    calculateTrend(dataPoints) {
      if (dataPoints.length < 2) return { direction: 'stable', strength: 0 };
      
      // 简单线性回归
      const n = dataPoints.length;
      const sumX = dataPoints.reduce((sum, _, i) => sum + i, 0);
      const sumY = dataPoints.reduce((sum, point) => sum + point.value, 0);
      const sumXY = dataPoints.reduce((sum, point, i) => sum + i * point.value, 0);
      const sumX2 = dataPoints.reduce((sum, _, i) => sum + i * i, 0);
      
      const slope = (n * sumXY - sumX * sumY) / (n * sumX2 - sumX * sumX);
      const intercept = (sumY - slope * sumX) / n;
      
      // 计算R²
      const yMean = sumY / n;
      const ssTotal = dataPoints.reduce((sum, point) => 
        sum + Math.pow(point.value - yMean, 2), 0
      );
      const ssResidual = dataPoints.reduce((sum, point, i) => 
        sum + Math.pow(point.value - (slope * i + intercept), 2), 0
      );
      const r2 = 1 - (ssResidual / ssTotal);
      
      return {
        direction: slope > 0.1 ? 'increasing' : slope < -0.1 ? 'decreasing' : 'stable',
        strength: Math.abs(slope),
        confidence: r2,
        slope,
        intercept,
        projection: (days) => slope * (n + days - 1) + intercept
      };
    }
  }

  class ExportService {
    constructor(analyticsService) {
      this.analyticsService = analyticsService;
      this.exportQueue = [];
      this.supportedFormats = ['json', 'csv', 'pdf', 'excel', 'html'];
    }
    
    async exportUserData(userId, format, options = {}) {
      const {
        startDate = new Date(Date.now() - 365 * 24 * 60 * 60 * 1000), // 默认1年
        endDate = new Date(),
        includeAnalytics = true,
        includeRawData = false,
        compress = false
      } = options;
      
      // 验证格式
      if (!this.supportedFormats.includes(format)) {
        throw new Error(`不支持的导出格式: ${format}`);
      }
      
      // 收集数据
      const exportData = await this.collectExportData(
        userId,
        startDate,
        endDate,
        includeAnalytics,
        includeRawData
      );
      
      // 根据格式生成导出文件
      let exportResult;
      switch (format) {
        case 'json':
          exportResult = await this.exportToJSON(exportData);
          break;
        case 'csv':
          exportResult = await this.exportToCSV(exportData);
          break;
        case 'pdf':
          exportResult = await this.exportToPDF(exportData);
          break;
        case 'excel':
          exportResult = await this.exportToExcel(exportData);
          break;
        case 'html':
          exportResult = await this.exportToHTML(exportData);
          break;
      }
      
      // 压缩处理
      if (compress) {
        exportResult = await this.compressExport(exportResult);
      }
      
      // 记录导出日志
      await this.logExport(userId, format, options, exportResult);
      
      return exportResult;
    }
    
    async collectExportData(userId, startDate, endDate, includeAnalytics, includeRawData) {
      const data = {
        metadata: {
          userId,
          exportDate: new Date(),
          dateRange: { startDate, endDate },
          version: '1.0.0'
        },
        summary: {},
        details: {}
      };
      
      // 收集基础数据
      const [user, mainChains, auxChains, patternTrees, activityLogs] = await Promise.all([
        User.findOne({ userId }).lean(),
        MainChain.find({ userId, createdAt: { $gte: startDate, $lte: endDate } }).lean(),
        AuxiliaryChain.find({ userId, createdAt: { $gte: startDate, $lte: endDate } }).lean(),
        PatternTree.find({ userId, createdAt: { $gte: startDate, $lte: endDate } }).lean(),
        ActivityLog.find({ userId, timestamp: { $gte: startDate, $lte: endDate } }).lean()
      ]);
      
      data.user = this.sanitizeUserData(user);
      
      if (includeRawData) {
        data.rawData = {
          mainChains,
          auxChains,
          patternTrees,
          activityLogs
        };
      }
      
      // 添加分析数据
      if (includeAnalytics) {
        const analytics = await this.analyticsService.getUserAnalytics(userId, {
          startDate,
          endDate,
          includeDetails: true
        });
        
        data.analytics = analytics;
        data.summary = this.generateExportSummary(analytics);
      }
      
      // 生成统计摘要
      data.statistics = {
        totalMainChains: mainChains.length,
        totalAuxChains: auxChains.length,
        totalPatternTrees: patternTrees.length,
        totalActivities: activityLogs.length,
        dataPoints: this.countDataPoints(data)
      };
      
      return data;
    }
    
    async exportToJSON(data) {
      const jsonString = JSON.stringify(data, null, 2);
      const buffer = Buffer.from(jsonString, 'utf-8');
      
      return {
        format: 'json',
        filename: `export_${data.metadata.userId}_${Date.now()}.json`,
        mimeType: 'application/json',
        size: buffer.length,
        buffer,
        checksum: this.calculateChecksum(buffer)
      };
    }
    
    async exportToCSV(data) {
      const csv = require('csv-writer');
      const streams = [];
      
      // 为不同类型的数据创建不同的CSV文件
      
      // 1. 任务数据CSV
      if (data.rawData?.mainChains) {
        const taskRecords = [];
        data.rawData.mainChains.forEach(chain => {
          chain.nodes.forEach(node => {
            taskRecords.push({

              nodeId: node.nodeId,
              startTime: node.startTime,
              endTime: node.endTime,
              duration: node.duration,
              status: node.status,
              description: node.taskDescription,
              violationCount: node.violationRules?.length || 0
            });
          });
        });
        
        const taskCSV = await this.createCSV(taskRecords, 'tasks');
        streams.push(taskCSV);
      }
      
      // 2. 定式数据CSV
      if (data.rawData?.patternTrees) {
        const patternRecords = [];
        data.rawData.patternTrees.forEach(tree => {
          tree.nodes.forEach(node => {
            patternRecords.push({
              treeId: tree.treeId,
              treeTitle: tree.title,
              currentState: tree.currentState,
              nodeId: node.nodeId,
              nodeTitle: node.title,
              level: node.level,
              status: node.status,
              successRate: node.stats.totalAttempts > 0 
                ? (node.stats.successCount / node.stats.totalAttempts).toFixed(2) 
                : 0,
              stabilityScore: node.stats.stabilityScore,
              addedDate: node.addedDate
            });
          });
        });
        
        const patternCSV = await this.createCSV(patternRecords, 'patterns');
        streams.push(patternCSV);
      }
      
      // 3. 分析摘要CSV
      if (data.analytics) {
        const summaryRecords = [{
          metric: '总任务数',
          value: data.analytics.overview.totalTasks
        }, {
          metric: '完成任务数',
          value: data.analytics.overview.completedTasks
        }, {
          metric: '任务成功率',
          value: (data.analytics.overview.taskSuccessRate * 100).toFixed(1) + '%'
        }, {
          metric: '总投入时间(分钟)',
          value: data.analytics.overview.totalTimeInvested
        }, {
          metric: '活跃天数',
          value: data.analytics.overview.activeDays
        }];
        
        const summaryCSV = await this.createCSV(summaryRecords, 'summary');
        streams.push(summaryCSV);
      }
      
      // 合并所有CSV文件
      return {
        format: 'csv',
        filename: `export_${data.metadata.userId}_${Date.now()}.zip`,
        mimeType: 'application/zip',
        files: streams,
        checksum: this.calculateChecksum(Buffer.concat(streams.map(s => s.buffer)))
      };
    }

    async cleanupOldData(userId = null, dryRun = false) {
      const results = {
        deletedRecords: {},
        freedSpace: 0,
        errors: [],
        timestamp: new Date()
      };
      
      try {
        // 清理活动日志
        const activityLogResult = await this.cleanupActivityLogs(userId, dryRun);
        results.deletedRecords.activityLogs = activityLogResult;
        
        // 清理过期的链数据
        const chainResult = await this.cleanupChains(userId, dryRun);
        results.deletedRecords.chains = chainResult;
        
        // 清理临时数据
        const tempResult = await this.cleanupTempData(userId, dryRun);
        results.deletedRecords.tempData = tempResult;
        
        // 清理导出缓存
        const exportResult = await this.cleanupExportCache(userId, dryRun);
        results.deletedRecords.exports = exportResult;
        
        // 计算释放的空间
        results.freedSpace = await this.calculateFreedSpace(results.deletedRecords);
        
        // 记录清理日志
        if (!dryRun) {
          await this.logService.log(
            userId || 'system',
            'data_cleanup',
            'execute',
            results
          );
        }
        
      } catch (error) {
        results.errors.push(error.message);
        await this.logService.logError('data_cleanup_error', error);
      }
      
      return results;
    }
    
    async archiveUserData(userId, options = {}) {
      const {
        includeAll = true,
        compress = true,
        encryption = false,
        password = null
      } = options;
      
      // 收集所有用户数据
      const userData = await this.collectAllUserData(userId);
      
      // 创建归档
      const archive = {
        metadata: {
          userId,
          archiveDate: new Date(),
          version: '1.0.0',
          recordCount: this.countRecords(userData)
        },
        data: userData
      };
      
      // 压缩处理
      let archiveBuffer = Buffer.from(JSON.stringify(archive));
      
      if (compress) {
        const zlib = require('zlib');
        archiveBuffer = await promisify(zlib.gzip)(archiveBuffer);
      }
      
      // 加密处理
      if (encryption && password) {
        archiveBuffer = await this.encryptData(archiveBuffer, password);
      }
      
      // 保存归档
      const archiveId = await this.saveArchive(userId, archiveBuffer, options);
      
      return {
        archiveId,
        size: archiveBuffer.length,
        compressed: compress,
        encrypted: encryption,
        timestamp: new Date()
      };
    }
