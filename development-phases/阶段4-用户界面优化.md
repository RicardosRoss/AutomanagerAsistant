## 开发目标

优化用户交互界面，完善命令行系统，实现智能化的用户引导和错误处理机制，提升整体用户体验，确保普通用户能够轻松掌握CTDP和RSIP协议的使用。

## 用户体验设计原则

### 1. 渐进式复杂度
- **新手模式**: 简化命令，自动推荐参数
- **专家模式**: 完整命令集，高级功能
- **智能切换**: 根据用户使用经验自动调整

### 2. 即时反馈原则
- **操作确认**: 每个动作都有明确的反馈
- **进度提示**: 长时间操作显示进度
- **状态同步**: 实时更新用户状态信息

### 3. 容错设计
- **预防错误**: 参数验证和提示
- **优雅降级**: 部分功能失败不影响核心功能
- **快速恢复**: 提供撤销和重试机制

  class CommandParserService {
    constructor() {
      this.commandTemplates = new Map();
      this.userContexts = new Map(); // 用户上下文缓存
      this.setupCommandTemplates();
    }
    
    setupCommandTemplates() {
      // CTDP命令模板
      this.commandTemplates.set('create_main', {
        pattern: /^\/create_main\s+(.+)$/,
        description: '创建主链',
        params: [
          { name: 'title', type: 'string', required: true, description: '主链标题' }
        ],
        examples: [
          '/create_main 学习链',
          '/create_main 工作专注链'
        ],
        hints: {
          beginner: '建议从简单的标题开始，如"每日学习"',
          advanced: '可以使用更具体的描述，如"深度工作-编程学习链"'
        }
      });
      
      this.commandTemplates.set('start_task', {
        pattern: /^\/start_task\s+(\w+)(?:\s+(\d+))?(?:\s+(.+))?$/,
        description: '开始任务',
        params: [
          { name: 'chainId', type: 'string', required: true, description: '链ID' },
          { name: 'duration', type: 'number', required: false, default: 60, description: '时长(分钟)' },
          { name: 'description', type: 'string', required: false, description: '任务描述' }
        ],
        examples: [
          '/start_task main_001 60 学习Python',
          '/start_task main_002 45',
          '/start_task main_003'
        ]
      });
      
      // RSIP命令模板
      this.commandTemplates.set('add_pattern', {
        pattern: /^\/add_pattern\s+(\w+)\s+(\w+|root)\s+"([^"]+)"(?:\s+"([^"]+)")?$/,
        description: '添加定式',
        params: [
          { name: 'treeId', type: 'string', required: true, description: '定式树ID' },
          { name: 'parentId', type: 'string', required: true, description: '父节点ID或root' },
          { name: 'title', type: 'string', required: true, description: '定式标题' },
          { name: 'rules', type: 'string', required: false, description: '定式规则' }
        ],
        examples: [
          '/add_pattern tree_001 root "不带手机进卧室" "睡前1小时不使用手机"',
          '/add_pattern tree_001 pattern_001 "固定起床时间"'
        ]
      });
    }
    
    async parseCommand(userId, message) {
      const text = message.text?.trim();
      if (!text?.startsWith('/')) {
        return await this.handleNaturalLanguage(userId, text);
      }
      
      const commandName = this.extractCommandName(text);
      const template = this.commandTemplates.get(commandName);
      
      if (!template) {
        return await this.handleUnknownCommand(userId, text);
      }
      
      const parseResult = await this.parseWithTemplate(userId, text, template);
      return parseResult;
    }
    
    async parseWithTemplate(userId, text, template) {
      const match = text.match(template.pattern);
      
      if (!match) {
        return {
          success: false,
          command: template.description,
          error: 'command_format_error',
          suggestion: await this.generateFormatSuggestion(userId, template),
          examples: template.examples
        };
      }
      
      const params = {};
      const errors = [];
      
      // 解析参数
      for (let i = 0; i < template.params.length; i++) {
        const param = template.params[i];
        const value = match[i + 1];
        
        if (!value && param.required) {
          errors.push(`缺少必需参数: ${param.name} (${param.description})`);
          continue;
        }
        
        if (value) {
          const validatedValue = await this.validateParam(param, value);
          if (validatedValue.valid) {
            params[param.name] = validatedValue.value;
          } else {
            errors.push(`参数 ${param.name} 无效: ${validatedValue.error}`);
          }
        } else if (param.default !== undefined) {
          params[param.name] = param.default;
        }
      }
      
      if (errors.length > 0) {
        return {
          success: false,
          command: template.description,
          error: 'parameter_validation_error',
          errors,
          suggestion: await this.generateParameterSuggestion(userId, template, params)
        };
      }
      
      return {
        success: true,
        command: this.extractCommandName(text),
        params,
        template
      };
    }
    
    async handleNaturalLanguage(userId, text) {
      // 自然语言理解和命令推荐
      const intent = await this.detectIntent(text);
      const recommendations = await this.generateCommandRecommendations(userId, intent);
      
      return {
        success: false,
        error: 'natural_language_input',
        intent,
        message: `我理解您想要${intent.description}，以下是相关命令:`,
        recommendations
      };
    }
    
    async detectIntent(text) {
      const intentPatterns = {
        create_main_chain: /创建|新建|开始.*(链|任务链|主链)/,
        start_task: /开始|启动.*(任务|工作|学习)/,
        create_pattern: /添加|创建|新增.*(定式|规则|模式)/,
        view_status: /查看|显示|状态|情况/,
        help: /帮助|指南|怎么|如何/
      };
      
      for (const [intent, pattern] of Object.entries(intentPatterns)) {
        if (pattern.test(text)) {
          return {
            type: intent,
            confidence: 0.8,
            description: this.getIntentDescription(intent)
          };
        }
      }
      
      return {
        type: 'unknown',
        confidence: 0.3,
        description: '执行某些操作'
      };
    }
    
    async generateCommandRecommendations(userId, intent) {
      const userContext = await this.getUserContext(userId);
      const recommendations = [];
      
      switch (intent.type) {
        case 'create_main_chain':
          recommendations.push({
            command: '/create_main',
            description: '创建新的主链',
            example: '/create_main 我的学习链',
            priority: 'high'
          });
          break;
          
        case 'start_task':
          if (userContext.hasMainChains) {
            const recentChain = userContext.recentMainChain;
            recommendations.push({
              command: '/start_task',
              description: '开始任务',
              example: `/start_task ${recentChain.chainId} 60 专注学习`,
              priority: 'high'
            });
          } else {
            recommendations.push({
              command: '/create_main',
              description: '先创建主链',
              example: '/create_main 学习链',
              priority: 'high',
              note: '需要先创建主链才能开始任务'
            });
          }
          break;
          
        case 'help':
          recommendations.push(
            {
              command: '/help',
              description: '查看帮助文档',
              priority: 'high'
            },
            {
              command: '/tutorial',
              description: '交互式教程',
              priority: 'medium'
            }
          );
          break;
      }
      
      return recommendations;
    }
  }

  class ErrorHandlingService {
    constructor(bot, logService) {
      this.bot = bot;
      this.logService = logService;
      this.errorPatterns = this.setupErrorPatterns();
      this.userErrorHistory = new Map(); // 用户错误历史
    }
    
    setupErrorPatterns() {
      return {
        // 命令相关错误
        COMMAND_NOT_FOUND: {
          pattern: /unknown command|未知命令/,
          severity: 'low',
          category: 'command',
          solutions: [
            '输入 /help 查看所有可用命令',
            '检查命令拼写是否正确',
            '使用 /tutorial 学习基本操作'
          ]
        },
        
        INVALID_PARAMETERS: {
          pattern: /invalid parameter|参数无效|parameter validation error/,
          severity: 'medium',
          category: 'input',
          solutions: [
            '检查参数格式是否正确',
            '确认必需参数都已提供',
            '参考命令示例重新输入'
          ]
        },
        
        // 业务逻辑错误
        CHAIN_NOT_FOUND: {
          pattern: /链不存在|chain not found/,
          severity: 'medium',
          category: 'business',
          solutions: [
            '使用 /chains 查看现有链',
            '检查链ID是否输入正确',
            '如果链被意外删除，可以重新创建'
          ]
        },
        
        TASK_ALREADY_RUNNING: {
          pattern: /任务正在运行|task already running/,
          severity: 'medium',
          category: 'state',
          solutions: [
            '等待当前任务完成',
            '使用 /status 查看任务状态',
            '如需中断，请先完成或取消当前任务'
          ]
        },
        
        // 系统错误
        DATABASE_ERROR: {
          pattern: /database|数据库.*错误/,
          severity: 'high',
          category: 'system',
          solutions: [
            '请稍后重试',
            '如问题持续，请联系技术支持',
            '临时解决方案: 重启机器人'
          ]
        },
        
        NETWORK_ERROR: {
          pattern: /network|网络.*错误/,
          severity: 'high',
          category: 'network',
          solutions: [
            '检查网络连接',
            '稍后重试操作',
            '如问题持续，可能是服务器维护'
          ]
        }
      };
    }
    
    async handleError(userId, chatId, error, context = {}) {
      const errorInfo = await this.analyzeError(error);
      const userHistory = this.getUserErrorHistory(userId);
      
      // 记录错误
      await this.logError(userId, error, context, errorInfo);
      
      // 更新用户错误历史
      this.updateUserErrorHistory(userId, errorInfo);
      
      // 生成个性化错误响应
      const response = await this.generateErrorResponse(
        userId,
        errorInfo,
        userHistory,
        context
      );
      
      // 发送错误消息
      await this.sendErrorMessage(chatId, response);
      
      // 如果是重复错误，提供额外帮助
      if (this.isRepeatedError(userHistory, errorInfo)) {
        await this.offerAdditionalHelp(chatId, userId, errorInfo);
      }
      
      return errorInfo;
    }
    
    analyzeError(error) {
      const errorMessage = error.message || error.toString();
      let category = 'unknown';
      let severity = 'medium';
      let solutions = ['请稍后重试'];
      let errorType = 'UNKNOWN_ERROR';
      
      // 匹配错误模式
      for (const [type, pattern] of Object.entries(this.errorPatterns)) {
        if (pattern.pattern.test(errorMessage)) {
          category = pattern.category;
          severity = pattern.severity;
          solutions = pattern.solutions;
          errorType = type;
          break;
        }
      }
      
      return {
        type: errorType,
        message: errorMessage,
        category,
        severity,
        solutions,
        timestamp: new Date(),
        originalError: error
      };
    }
    
    async generateErrorResponse(userId, errorInfo, userHistory, context) {
      const isNewUser = userHistory.length < 5;
      const hasRepeatedSimilarErrors = userHistory
        .filter(h => h.category === errorInfo.category).length > 2;
      
      let response = {
        title: this.getErrorTitle(errorInfo),
        message: this.getErrorMessage(errorInfo, isNewUser),
        solutions: this.getPersonalizedSolutions(errorInfo, isNewUser, hasRepeatedSimilarErrors),
        severity: errorInfo.severity,
        quickActions: await this.generateQuickActions(userId, errorInfo, context)
      };
      
      // 为新用户提供更详细的说明
      if (isNewUser) {
        response.beginnerNote = '💡 作为新用户，建议您先完成新手教程以熟悉系统操作';
        response.quickActions.unshift({
          text: '📚 开始教程',
          callback_data: 'start_tutorial'
        });
      }
      
      return response;
    }
    
    async sendErrorMessage(chatId, response) {
      const { title, message, solutions, quickActions, beginnerNote, severity } = response;
      
      const severityEmoji = {
        low: '⚠️',
        medium: '❌',
        high: '🚨'
      };
      
      let text = `${severityEmoji[severity]} **${title}**\n\n`;
      text += `${message}\n\n`;
      
      if (solutions.length > 0) {
        text += `💡 **解决方案**:\n`;
        solutions.forEach((solution, index) => {
          text += `${index + 1}. ${solution}\n`;
        });
        text += '\n';
      }
      
      if (beginnerNote) {
        text += `${beginnerNote}\n\n`;
      }
      
      text += `❓ 如仍有问题，请使用 /help 或 /support`;
      
      const keyboard = quickActions.length > 0 ? {
        reply_markup: {
          inline_keyboard: this.formatQuickActions(quickActions)
        }
      } : {};
      
      await this.bot.sendMessage(chatId, text, {
        parse_mode: 'Markdown',
        ...keyboard
      });
    }
    
    generateQuickActions(userId, errorInfo, context) {
      const actions = [];
      
      switch (errorInfo.category) {
        case 'command':
          actions.push(
            { text: '📖 查看帮助', callback_data: 'show_help' },
            { text: '💬 命令示例', callback_data: 'show_examples' }
          );
          break;
          
        case 'input':
          if (context.lastCommand) {
            actions.push(
              { text: '🔧 修正参数', callback_data: `fix_params_${context.lastCommand}` },
              { text: '📋 参数说明', callback_data: `param_help_${context.lastCommand}` }
            );
          }
          break;
          
        case 'business':
          actions.push(
            { text: '📊 查看状态', callback_data: 'show_status' },
            { text: '🔄 刷新数据', callback_data: 'refresh_data' }
          );
          break;
          
        case 'system':
          actions.push(
            { text: '🔄 重试操作', callback_data: 'retry_last_operation' },
            { text: '🆘 联系支持', callback_data: 'contact_support' }
          );
          break;
      }
      
      // 通用操作
      actions.push(
        { text: '🏠 返回主页', callback_data: 'go_home' }
      );
      
      return actions;
    }
    
    async offerAdditionalHelp(chatId, userId, errorInfo) {
      const helpMessage = `🤔 看起来您在 ${errorInfo.category} 方面遇到了重复问题\n\n` +
        `我为您准备了一些额外的帮助资源:`;
      
      const helpActions = [
        { text: '📹 观看视频教程', callback_data: `video_help_${errorInfo.category}` },
        { text: '📚 详细文档', callback_data: `docs_${errorInfo.category}` },
        { text: '👨‍💼 人工客服', callback_data: 'human_support' },
        { text: '❓ 常见问题', callback_data: `faq_${errorInfo.category}` }
      ];

        reply_markup: {
          inline_keyboard: this.formatQuickActions(helpActions)
        }
      });
    }
    
    // 智能重试机制
    async handleRetryOperation(userId, chatId, operationData) {
      const retryHistory = this.getRetryHistory(userId);
      const maxRetries = 3;
      
      if (retryHistory.count >= maxRetries) {
        await this.bot.sendMessage(
          chatId,
          `🚫 重试次数过多\n\n` +
          `为避免重复错误，建议您:\n` +
          `1. 检查输入参数\n` +
          `2. 查看系统状态\n` +
          `3. 联系技术支持\n\n` +
          `/support - 获取人工帮助`,
          {
            reply_markup: {
              inline_keyboard: [[
                { text: '🆘 联系支持', callback_data: 'contact_support' },
                { text: '🏠 返回主页', callback_data: 'go_home' }
              ]]
            }
          }
        );
        return;
      }
      
      try {
        // 执行重试逻辑
        await this.executeRetryOperation(operationData);
        
        // 重试成功，清除重试历史
        this.clearRetryHistory(userId);
        
        await this.bot.sendMessage(
          chatId,
          `✅ 重试成功！操作已完成`
        );
        
      } catch (error) {
        // 重试失败，增加计数
        this.incrementRetryHistory(userId);
        
        await this.handleError(userId, chatId, error, {
          isRetry: true,
          retryCount: retryHistory.count + 1,
          originalOperation: operationData
        });
      }
    }
  }

// components/UIComponents.js
class UIComponents {
  static createProgressBar(current, total, width = 20) {
    const filled = Math.round((current / total) * width);
    const empty = width - filled;
    const bar = '█'.repeat(filled) + '░'.repeat(empty);
    const percentage = Math.round((current / total) * 100);
    
    return `${bar} ${percentage}% (${current}/${total})`;
  }
  
  static createStatusCard(title, data, options = {}) {
    const { showTimestamp = true, emoji = '📊' } = options;
    
    let card = `${emoji} **${title}**\n\n`;
    
    Object.entries(data).forEach(([key, value]) => {
      const formattedKey = key.replace(/([A-Z])/g, ' $1').toLowerCase();
      card += `• ${formattedKey}: ${value}\n`;
    });
    
    if (showTimestamp) {
      card += `\n⏰ 更新时间: ${new Date().toLocaleString('zh-CN')}`;
    }
    
    return card;
  }
  
  static createActionButtons(actions, maxPerRow = 2) {
    const rows = [];
    for (let i = 0; i < actions.length; i += maxPerRow) {
      rows.push(actions.slice(i, i + maxPerRow));
    }
    return { inline_keyboard: rows };
  }
  
  static formatDuration(seconds) {
    const hours = Math.floor(seconds / 3600);
    const minutes = Math.floor((seconds % 3600) / 60);
    const secs = seconds % 60;
    
    if (hours > 0) {
      return `${hours}时${minutes}分${secs}秒`;
    } else if (minutes > 0) {
      return `${minutes}分${secs}秒`;
    } else {
      return `${secs}秒`;
    }
  }
}
