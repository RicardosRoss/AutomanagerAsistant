      await this.warmup();
      
      // 收集基准内存使用
      global.gc();
      const baselineMemory = process.memoryUsage();
      heapSnapshots.push(baselineMemory);
      
      // 执行重复操作
      for (let i = 0; i < iterations; i++) {
        await this.executeOperation();
        
        if (i % 10 === 0) {
          global.gc();
          heapSnapshots.push(process.memoryUsage());
        }
      }
      
      // 分析内存增长
      const analysis = this.analyzeMemoryGrowth(heapSnapshots);
      
      return {
        hasLeak: analysis.growthRate > 0.1, // 10%以上的增长视为泄漏
        growthRate: analysis.growthRate,
        details: analysis
      };
    }
    
    async executeOperation() {
      // 执行可能导致内存泄漏的操作
      const operations = [
        () => this.createAndDeleteChain(),
        () => this.startAndCompleteTask(),
        () => this.generateAnalytics(),
        () => this.exportData()
      ];
      
      for (const op of operations) {
        await op();
      }

    analyzeMemoryGrowth(snapshots) {
      const heapUsed = snapshots.map(s => s.heapUsed);
      const firstSnapshot = heapUsed[0];
      const lastSnapshot = heapUsed[heapUsed.length - 1];
      
      const growthRate = (lastSnapshot - firstSnapshot) / firstSnapshot;
      
      // 线性回归分析趋势
      const trend = this.calculateTrend(heapUsed);
      
      return {
        growthRate,
        trend,
        initialMemory: firstSnapshot,
        finalMemory: lastSnapshot,
        peakMemory: Math.max(...heapUsed)
      };
    }
  }
